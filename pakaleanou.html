function initMap() { const map = new google.maps.Map(document.getElementById("map"), { zoom: 12, center: { lat: 38.1356291, lng: 23.8257552 } // Κέντρο στην πρώτη τοποθεσία }); const locations = [ { lat: 38.1356291, lng: 23.8257552, title: "Νήσων 5, Κρυονέρι", label: "7. 319376" }, { address: "Παλλήνης 19Β, Κηφισιά", title: "Παλλήνης 19Β, Κηφισιά", label: "10. 400553" } ]; let currentInfoWindow = null; // Συνάρτηση για geocoding function geocodeAddress(address, callback) { const geocoder = new google.maps.Geocoder(); geocoder.geocode({ address: address }, (results, status) => { if (status === "OK") { callback(results[0].geometry.location); } else { console.error("Geocoding failed for address " + address + ": " + status); callback(null); } }); } // Προσθήκη πινέζων και συμπλήρωση πίνακα locations.forEach((location, index) => { // Λειτουργία για προσθήκη πινέζας const addMarker = (coords) => { if (coords) { const marker = new google.maps.Marker({ position: coords, map: map, title: location.title, label: { text: location.label, color: "black", fontSize: "14px", fontWeight: "bold" } }); const infoWindow = new google.maps.InfoWindow({ content: `

${location.title}

` }); marker.addListener("click", () => { if (currentInfoWindow) { currentInfoWindow.close(); } infoWindow.open(map, marker); currentInfoWindow = infoWindow; document.querySelectorAll("#table-body tr").forEach(row => { row.classList.remove("highlighted"); }); const row = document.getElementById(`row-${location.label}`); if (row) { row.classList.add("highlighted"); } }); map.addListener("click", () => { if (currentInfoWindow) { currentInfoWindow.close(); currentInfoWindow = null; } document.querySelectorAll("#table-body tr").forEach(row => { row.classList.remove("highlighted"); }); }); } }; // Έλεγχος αν υπάρχει lat/lng ή address if (typeof location.lat === 'number' && location.lat !== null && typeof location.lng === 'number' && location.lng !== null) { addMarker({ lat: location.lat, lng: location.lng }); } else if (location.address && typeof location.address === 'string') { geocodeAddress(location.address, addMarker); } // Συμπλήρωση του πίνακα const tableBody = document.getElementById("table-body"); const row = document.createElement("tr"); row.id = `row-${location.label}`; row.innerHTML = ` ${location.label} ${location.title} ${ typeof location.lat === 'number' && location.lat !== null && typeof location.lng === 'number' && location.lng !== null ? `${location.lat}, ${location.lng}` : location.address && typeof location.address === 'string' ? location.address : 'N/A' } `; tableBody.appendChild(row); }); }
